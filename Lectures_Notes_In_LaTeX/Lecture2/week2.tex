\section{Variable Naming Conventions}
\textbf{Camel Case:}
\begin{itemize}

\end{itemize}

\textbf{Named Constants:}
\begin{itemize}
    \item Constants should be named in uppercase.
    \item Example: \texttt{final double HST = 1.13;}

\section{Error Types}
